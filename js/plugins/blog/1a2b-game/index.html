1A2B 猜數字遊戲-程式碼

import random

num = int(input("請輸入您想猜幾個數字的密碼："))
pwd = random.sample(range(0, 10), num)

A = 0
B = 0

while A!=num:
    guess = input("請輸入"+str(num)+"個數字：")
    while len(guess)!=num or len(set(guess))!=num:
        guess = input("請輸入"+str(num)+"個數字，且數字不重複：")
    list1 = list(map(int, list(guess)))
    A = 0  # 要先清空!再重計算
    B = 0  # 要先清空!再重計算
    for i in list1:   # 把list1裡的元素內容逐一拿出來給i
        if i in pwd:
            if list1.index(i) == pwd.index(i):
                A += 1
            else:
                B += 1
    print(guess, "：", A, "A", B, "B", sep="")

print("你猜對了，密碼是：", guess)
